<div id="photo_gallery">
  <h2>Photo Gallery</h2>
  {% for photo in page.photos %}
    <img src="{{ page.photos_base_url }}{{ photo }}"/>
  {% endfor %}
</div>
<script type="text/javascript">
  $(function() {
    $("#photo_gallery img").each(function() {
      if ($(this).width() > 590) { // for browsers that don't support max-width css
        $(this).width(590);
      }
    });
    if ($("#photo_gallery img").length > 4) {
      $("#photo_gallery img").addClass("thumbnail").click(function() {
        var big = $("#photo_gallery img:not(img.thumbnail)");
        if (!big.length) {
          big = $("<img style='width: 590px;'/>");
          big.prependTo("#photo_gallery");
        }
        big.attr('src', $(this).attr('src'));
      }).filter(":first").click();
    }
  });
</script>
